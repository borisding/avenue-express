import { Router } from 'express';

export function index(req, res, next) {
  // GET method, list all records
  next();
}

export function create(req, res, next) {
  // GET method, create new record
  next();
}

export function store(req, res, next) {
  // POST method, store new record
  next();
}

export function show(req, res, next) {
  // GET method, show particular record
  next();
}

export function edit(req, res, next) {
  // GET method, edit particular record
  next();
}

export function update(req, res, next) {
  // PUT method, update particular record
  next();
}

export function destroy(req, res, next) {
  // DELETE method, remove particular record
  next();
}

// respective routes based on the http verbs
// match the actions of generated resource controller, respectively
const router = Router();

router
  .get('/<path>', index)
  .get('/<path>/create', create)
  .get('/<path>/:id', show)
  .get('/<path>/:id/edit', edit)
  .post('/<path>', store)
  .put('/<path>/:id', update)
  .delete('/<path>/:id', destroy);

export default router;

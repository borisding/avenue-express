import express from 'express';
import * as <controller> from '@controllers/<controller>';

const <router> = express.Router();

<router>
  .get('/', <controller>.index)
  .get('/<segment>/create', <controller>.create)
  .get('/<segment>/:id', <controller>.show)
  .get('/<segment>/:id/edit', <controller>.edit)
  .post('/<segment>', <controller>.store)
  .put('/<segment>/:id', <controller>.update)
  .delete('/<segment>/:id', <controller>.remove);

export default <router>;
